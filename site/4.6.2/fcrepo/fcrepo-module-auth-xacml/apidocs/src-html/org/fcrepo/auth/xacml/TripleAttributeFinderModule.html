059/**
060 * Finds resource attributes based on triples in the Fedora graph. Retrieves values where the attribute URI matches the
061 * triple predicate and the triple object can be supplied as the requested data type.
062 *
063 * @author Gregory Jansen
064 * @author Andrew Woods
065 * @author Scott Prater
066 */
067@Component("tripleAttributeFinderModule")
068public class TripleAttributeFinderModule extends AttributeFinderModule {
069
070    private static final Logger LOGGER = getLogger(TripleAttributeFinderModule.class);
071
072    private static BagAttribute empty_bag;
073
074    private static IdentifierConverter<Resource, FedoraResource> idTranslator;
075
076    /**
077     * Fedora's ModeShape session factory.
078     */
079    @Inject
080    protected SessionFactory sessionFactory;
081
082    @Inject
083    protected NodeService nodeService;
084
085    /**
086     * Supported designator types.
087     */
088    private static final Set<Integer> DESIGNATOR_TYPES = unmodifiableSet(singleton(RESOURCE_TARGET));
089
090    /**
091     * Supports designators.
092     *
093     * @return if designator is supported.
094     * @see org.jboss.security.xacml.sunxacml.finder.AttributeFinderModule#isDesignatorSupported()
095     */
096    @Override
097    public final boolean isDesignatorSupported() {
098        return true;
099    }
100
101    /**
102     * Supports resource attributes.
103     *
104     * @return the supported designator types.
105     * @see org.jboss.security.xacml.sunxacml.finder.AttributeFinderModule#getSupportedDesignatorTypes()
106     */
107    @Override
108    public final Set<Integer> getSupportedDesignatorTypes() {
109        return DESIGNATOR_TYPES;
110    }
111
112    /**
113     * Finds the matching triples values.
114     *
115     * @param attributeId The URI of the attribute key (the predicate of the triple)
116     * @see org.jboss.security.xacml.sunxacml.finder.AttributeFinderModule#findAttribute (java.net.URI, java.net.URI,
117     *      java.net.URI, java.net.URI, org.jboss.security.xacml.sunxacml.EvaluationCtx, int)
118     */
119    @Override
120    public final EvaluationResult findAttribute(final URI attributeType,
121                                                final URI attributeId,
122                                                final URI issuer,
123                                                final URI subjectCategory,
124                                                final EvaluationCtx context,
125                                                final int designatorType) {
126        LOGGER.debug("findAttribute({}, {}, {}, {}, {}, {})",
127                     attributeType, attributeId, issuer, subjectCategory, context, designatorType);
128
129        empty_bag = createEmptyBag(attributeType);
130
131        // Make sure this is a Resource attribute
132        if (designatorType != RESOURCE_TARGET) {
133            LOGGER.debug("Not looking for a resource attribute");
134            return new EvaluationResult(empty_bag);
135        }
136
137        final Session session;
138        try {
139            session = sessionFactory.getInternalSession();
140        } catch (final RepositoryRuntimeException e) {
141            LOGGER.debug("Error getting session!");
142            final Status status = new Status(singletonList(STATUS_PROCESSING_ERROR), "Error getting session");
143            return new EvaluationResult(status);
144        }
145
146        // The resourceId is the path of the object be acted on, retrieved from the PDP evaluation context
147        final EvaluationResult ridEvalRes =
148                context.getResourceAttribute(URI.create("http://www.w3.org/2001/XMLSchema#string"),
149                        URIConstants.ATTRIBUTEID_RESOURCE_ID, null);
150        final AttributeValue resourceIdAttValue = ridEvalRes.getAttributeValue();
151        if (resourceIdAttValue.getValue().toString().isEmpty()) {
152            LOGGER.debug("Context should have a resource-id attribute!");
153            final Status status = new Status(singletonList(STATUS_PROCESSING_ERROR), "Resource Id not found!");
154            return new EvaluationResult(status);
155        }
156
157        String resourceId = (String) resourceIdAttValue.getValue();
158
159        // if dealing with set_property action, use parent node for triples
160        final Set<String> actions = PolicyUtil.getActions(context);
161        if (actions.contains("set_property") || actions.contains("add_node")) {
162            final int index = resourceId.lastIndexOf("/{");
163            if (index > -1) {
164                resourceId = resourceId.substring(0, index);
165            }
166
167            if (resourceId.isEmpty()) {
168                resourceId = "/";
169            }
170        }
171
172        // Get the resource to be acted on
173        final FedoraResource resource;
174        final String path;
175        try {
176            resource = nodeService.find(session, resourceId);
177            if (resource == null) {
178                LOGGER.debug("Cannot find a fedora resource for {}", resourceId);
179                return new EvaluationResult(empty_bag);
180            }
181            path = resource.getPath();
182            idTranslator = new DefaultIdentifierTranslator(session);
183
184        } catch (final RepositoryRuntimeException e) {
185            // If the object does not exist, it may be due to the action being "create"
186            return new EvaluationResult(empty_bag);
187        }
188
189        LOGGER.debug("Looking for properties on modeshape path {} with repo path {}", resourceId, path);
190
191        // Get the properties of the resource
192        Model properties;
193        try {
194            properties = resource.getTriples(idTranslator, PROPERTIES).collect(toModel());
195
196        } catch (final RepositoryRuntimeException e) {
197            LOGGER.debug("Cannot retrieve any properties for [{}]:  {}", resourceId, e);
198            final Status status =
199                    new Status(singletonList(STATUS_PROCESSING_ERROR),
200                               "Error retrieving properties for [" + path + "]!");
201            return new EvaluationResult(status);
202        }
203
204        final Resource graphNode = idTranslator.reverse().convert(resource);
205        if (null == graphNode) {
206            LOGGER.debug("Cannot get subject for[{}]", resource.getPath());
207            final Status status =
208                    new Status(singletonList(STATUS_PROCESSING_ERROR),
209                            "Error retrieving properties for [" + path + "]!");
210            return new EvaluationResult(status);
211        }
212
213        LOGGER.debug("Looking for properties on graph node: {}", graphNode.getURI());
214
215        // Get the values of the properties matching the type
216        final Iterator<RDFNode> matches =
217                properties.listObjectsOfProperty(graphNode, properties.createProperty(attributeId.toString()));
218
219        final Set<AttributeValue> attr_bag = new HashSet<>();
220
221        // Add the properties to the bag
222        while (matches.hasNext()) {
223            final RDFNode match = matches.next();
224            final String uri = match.asResource().getURI();
225            LOGGER.debug("Found property: {}", uri);
226            attr_bag.add(new AnyURIAttribute(URI.create(uri)));
227        }
228
229        // Return the results, or any empty bag
230        if (attr_bag.isEmpty()) {
231            LOGGER.debug("No matching properties found");
232            return new EvaluationResult(empty_bag);
233        }
234
235        return new EvaluationResult(new BagAttribute(attributeType, attr_bag));
236    }
