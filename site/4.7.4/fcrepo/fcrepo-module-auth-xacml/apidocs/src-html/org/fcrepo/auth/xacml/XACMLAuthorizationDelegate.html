040/**
041 * Responsible for resolving Fedora's permissions within ModeShape via a XACML
042 * Policy Decision Point (PDP).
043 *
044 * @author Gregory Jansen
045 */
046@Component("fad")
047@Deprecated
048public class XACMLAuthorizationDelegate extends AbstractRolesAuthorizationDelegate {
049
050    public static final String EVERYONE_NAME = "EVERYONE";
051
052    /**
053     * The security principal for every request, that represents the "EVERYONE" user.
054     */
055    private static final Principal EVERYONE = new Principal() {
056
057        // Currently, this is identical to the EVERYONE principal defined in the RBACL module. This is done to
058        // preserve compatibility with earlier versions, where the definition of EVERYONE was part of the
059        // ServletContainerAuthenticationProvider and shared with all of the authorization modules. Currently, the
060        // XACML module does not appear to actually use this principal, and it may be worth reviewing in future for
061        // removal, or for changing it to a more XACML-specific concept of "everyone".
062
063        @Override
064        public String getName() {
065            return XACMLAuthorizationDelegate.EVERYONE_NAME;
066        }
067
068        @Override
069        public String toString() {
070            return getName();
071        }
072
073    };
074
075    /**
076     * Class-level logger.
077     */
078    private static final Logger LOGGER = LoggerFactory.getLogger(XACMLAuthorizationDelegate.class);
079
080    @Inject
081    private PDPFactory pdpFactory;
082
083    /**
084     * The XACML PDP.
085     */
086    private PDP pdp = null;
087
088    /**
089     * The standard environment attribute finder, supplies date/time.
090     */
091    private final CurrentEnvModule currentEnvironmentAttributeModule = new CurrentEnvModule();
092
093    /**
094     * The triple-based resource attribute finder module.
095     */
096    @Inject
097    private TripleAttributeFinderModule tripleResourceAttributeFinderModule;
098
099    /**
100     * The SPARQL-based resource attribute finder module.
101     */
102    @Inject
103    private SparqlResourceAttributeFinderModule sparqlResourceAttributeFinderModule;
104
105    /**
106     * Configures the delegate.
107     */
108    @PostConstruct
109    public final void init() {
110        pdp = pdpFactory.makePDP();
111        if (pdp == null) {
112            throw new Error("There is no PDP wired by the factory in the Spring context.");
113        }
114    }
115
116    /*
117     * (non-Javadoc)
118     * @see
119     * org.fcrepo.auth.common.FedoraAuthorizationDelegate#hasPermission(javax
120     * .jcr.Session, org.modeshape.jcr.value.Path, java.lang.String[])
121     */
122    @Override
123    public boolean rolesHavePermission(final Session session,
124                                       final String absPath,
125                                       final String[] actions,
126                                       final Set<String> roles) {
127        LOGGER.warn("===========================");
128        LOGGER.warn("This authorization provider is deprecated and will be removed in a future release of Fedora: {}",
129                this.getClass());
130        LOGGER.warn("===========================");
131
132        final EvaluationCtx evaluationCtx = buildEvaluationContext(session, absPath, actions, roles);
133        final ResponseCtx resp = pdp.evaluate(evaluationCtx);
134
135        boolean permit = true;
136        for (final Object o : resp.getResults()) {
137            final Result res = (Result) o;
138            if (LOGGER.isDebugEnabled()) {
139                try (ByteArrayOutputStream baos = new ByteArrayOutputStream()) {
140                    res.encode(baos);
141                    LOGGER.debug("ResponseCtx dump:\n{}", baos.toString("utf-8"));
142                } catch (final IOException e) {
143                    LOGGER.info("Cannot print response context", e);
144                }
145            }
146            if (Result.DECISION_PERMIT != res.getDecision()) {
147                permit = false;
148                break;
149            }
150        }
151
152        LOGGER.debug("Request for actions: {}, on path: {}, with roles: {}. Permission={}",
153                     actions,
154                     absPath,
155                     roles,
156                     permit);
157        return permit;
158    }
159
160    /**
161     * Builds a global attribute finder from injected modules that may use
162     * current session information.
163     *
164     * @param session the ModeShape session
165     * @param absPath the node or property path
166     * @param actions the actions requested
167     * @return an attribute finder
168     */
169    private EvaluationCtx buildEvaluationContext(final Session session,
170                                                 final String absPath,
171                                                 final String[] actions,
172                                                 final Set<String> roles) {
173        final FedoraEvaluationCtxBuilder builder = new FedoraEvaluationCtxBuilder();
174        builder.addFinderModule(currentEnvironmentAttributeModule);
175        builder.addFinderModule(sparqlResourceAttributeFinderModule);
176
177        // A subject attribute finder prototype is injected with Session
178        // AttributeFinderModule subjectAttributeFinder = null;
179        // if (applicationContext
180        // .containsBeanDefinition(SUBJECT_ATTRIBUTE_FINDER_BEAN)) {
181        // subjectAttributeFinder =
182        // (AttributeFinderModule) applicationContext.getBean(
183        // SUBJECT_ATTRIBUTE_FINDER_BEAN, session);
184        // builder.addFinderModule(subjectAttributeFinder);
185        // }
186
187        // environment attribute finder is injected with Session
188        // AttributeFinderModule environmentAttributeFinder = null;
189        // if (applicationContext
190        // .containsBeanDefinition(ENVIRONMENT_ATTRIBUTE_FINDER_BEAN)) {
191        // environmentAttributeFinder =
192        // (AttributeFinderModule) applicationContext.getBean(
193        // ENVIRONMENT_ATTRIBUTE_FINDER_BEAN, session);
194        // builder.addFinderModule(environmentAttributeFinder);
195        // }
196
197        // Triple attribute finder will look in modeshape for any valid
198        // predicate URI, therefore it falls last in this list.
199        builder.addFinderModule(tripleResourceAttributeFinderModule);
200        LOGGER.debug("effective roles: {}", roles);
201
202        final Principal user = (Principal) session.getAttribute(FEDORA_USER_PRINCIPAL);
203        builder.addSubject(user.getName(), roles);
204        builder.addResourceID(absPath);
205        builder.addWorkspace(session.getWorkspace().getName());
206        builder.addActions(actions);
207
208        // add the original IP address
209        final HttpServletRequest request = (HttpServletRequest) session.getAttribute(FEDORA_SERVLET_REQUEST);
210        builder.addOriginalRequestIP(request.getRemoteAddr());
211
212        // add user's groups
213        @SuppressWarnings("unchecked")
214        final Set<Principal> allGroups = (Set<Principal>) session.getAttribute(FEDORA_ALL_PRINCIPALS);
215        LOGGER.debug("effective groups: {}", allGroups);
216        builder.addGroups(user, allGroups);
217
218        return builder.build();
219    }
220
221    /**
222     * Get the principal that represents the "EVERYONE" user.
223     */
224    @Override
225    public Principal getEveryonePrincipal() {
226        return EVERYONE;
227    }
